function for finding expected k at different beta (both simulated and theoretical)
function [vt, vi] = expected(b)
%parameters
alpha = 0.36;
beta = b;
sig = 0.01;
 
%input parameters
N = 20;
mu = (1/(1-alpha))*log(alpha*beta);
rho = alpha;
sigma = sig;
m = 3;
 
%Tauchen
[Z, Zprob] = tauchen(20,mu,rho, sigma,m);
 
M = Zprob - eye(20);
M(:, 20) = ones(20,1);
 
c = zeros(20,1);
c(20) = 1;
 
pi = linsolve(M', c);
vt = dot(exp(1).^Z, pi);
vi = exp(1)^(mu);
%+sig^2/(2*(1-alpha^2))
end
